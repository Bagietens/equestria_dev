country_event = {
	id = stalliongrad.12
	immediate = { log = "[GetDateText]: [Root.GetName]: event stalliongrad.12" }
	title = stalliongrad.12.t
	desc = stalliongrad.12.d
	picture = GFX_event_stg_red_army

	is_triggered_only = yes

	option = {
		name = stalliongrad.12.a
		trigger = { 
			check_variable = {
				var = STG_opposition_support_v
				value = 20
				compare = less_than
			}
		}
		immediate = { log = "[GetDateText]: [Root.GetName]: event stalliongrad.12.a" }
		hidden_effect = {
			set_country_flag = STG_flag_unity_in_party
		}
	}
	option = {
		name = stalliongrad.12.b
		trigger = { 
			AND = {
				check_variable = {
					var = STG_opposition_support_v
					value = 20
					compare = greater_than_or_equals
				}
				check_variable = {
					var = STG_opposition_support_v
					value = 50
					compare = less_than
				}
			}
		}
		immediate = { log = "[GetDateText]: [Root.GetName]: event stalliongrad.12.b" }
		hidden_effect = {
			set_country_flag = STG_flag_status_quo
		}
	}
